2-dimentional REAL-TO-COMPLEX FFT using Radix-2,3,4,5,8 FFT routine. Calculates on each last 2-dimention. This routine is FORWARD transform.

Parameters:

  • narray (Numo::DFloat) —

    >=2-dimentional DFloat NArray with shape = [.., ny, nx] where nx = (2**ip) * (3**iq) * (5**ir), ny = (2**jp) * (3**jq) * (5**jr).

Returns:

  • (Numo::DComplex) —

    Result DComplex NArray with shape = [.., ny, nx/2+1]

Raises:

  • (Numo::FFTE::RadixError) —

    if nx, ny is not (2^p)*(3^q)*(5^r).

static VALUE
numo_ffte_dzfft2d(int argc, VALUE *args, VALUE mod)
{
    narray_t *na;
    VALUE viopt;
    VALUE vb, vna;
    VALUE vres;
    int ndim;
    integer iopt=-1, iopt_zero=0;
    fft_opt_t *g;
    integer n1,n2;
    size_t n=1;
    size_t shape[2];
    ndfunc_arg_in_t ain[1] = {{cDF,2}};
    ndfunc_arg_out_t aout[1] = {{cDC,2,shape}};
    ndfunc_t ndf = { iter_fft_dzfft2d, NO_LOOP, 1, 1, ain, aout };

    switch(rb_scan_args(argc, args, "11", &vna, &viopt)) {
    case 2:
        iopt = NUM2INT(viopt);
    }
    GetNArray(vna,na);
    ndim = NA_NDIM(na);
    if (ndim<2) {
        rb_raise(eRadixError,"ndim(=%d) should >= 2",ndim);
    }

    n1 = NA_SHAPE(na)[NA_NDIM(na)-1];
    n *= n1;

    shape[1] = n1/2+1;

    if (!is235radix(n1)) {
        rb_raise(eRadixError,"%d-th dim length(=%ld) is not 2,3,5-radix",
                 NA_NDIM(na)-1,shape[1]);
    }

    n2 = NA_SHAPE(na)[NA_NDIM(na)-2];
    n *= n2;

    shape[0] = n2;

    if (!is235radix(n2)) {
        rb_raise(eRadixError,"%d-th dim length(=%ld) is not 2,3,5-radix",
                 NA_NDIM(na)-2,shape[0]);
    }


    vna = na_copy(vna);
    g = alloc_fft_opt(n, iopt, &vb);

    dzfft2d_(NULL, &n1,&n2, &iopt_zero, g->buf);
    vres = na_ndloop3(&ndf, g, 1, vna);
    RB_GC_GUARD(vb);

    return vres;
}
